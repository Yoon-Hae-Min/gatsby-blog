м§„н–ү лҸҷкё°

Breakingн”„лЎңм қнҠёлҘј м§„н–үн•ҳл©ҙм„ң м„ұлҠҘкіј мҪ”л“ң кө¬мЎ°м—җ лҢҖн•ҙм„ң м•„мү¬мҡҙ л¶Җл¶„мқҙ лӮЁм•ҳкі  мқҙм—җ мӢңк°„мқҙ м—Ҷм–ҙм„ң мҲҳм •н•ҳм§Җ лӘ»н•ң 2к°Җм§ҖлҘј лҰ¬нҺҷнҶ л§Ғ н”„лЎңм қнҠёлҘј нҶөн•ҙ мҲҳм •н•ҙ ліҙкі мһҗ н•ҳмҳҖлӢӨ мІ« лҰ¬нҺҷнҶ л§ҒмқҖ мқён”јлӢҲнӢ° мҠӨнҒ¬лЎӨмқ„ лҰ¬нҺҷнҶ л§ҒмқҙлӢӨ. (мқён”јлӢҲнӢ° мҠӨнҒ¬лЎӨ мөңм Ғнҷ”н•ҳкё° PR)

лҰ¬нҺҷнҶ л§Ғ мӢңлҰ¬мҰҲ

  1. вӯҗ мқён”јлӢҲнӢ° мҠӨнҒ¬лЎӨм—җм„ң лҚ°мқҙн„°к°Җ л§Һм•„м§Җл©ҙ н”„л Ҳмһ„мқҙ л–Ём–ҙм§җ (нҳ„мһ¬ кёҖ) вӯҗ
  2. л§ҲмқҙнҺҳмқҙм§Җ нҢ”лЎңмҡ° нҢ”лЎңмӣҢ modalм°Ҫм—җм„ң лҚ°мқҙн„° лҸҷкё°нҷ” л°©лІ•

кё°мЎҙ мқён”јлӢҲнӢ° мҠӨнҒ¬лЎӨ м „лһө

кё°мЎҙм—җ мқён”јлӢҲнӢ° мҠӨнҒ¬лЎӨ л°©лІ•мқҖ IntersectionObserver к°қмІҙлҘј мқҙмҡ©н•ҙ мқён”јлӢҲнӢ° мҠӨнҒ¬лЎӨмқҙ н•„мҡ”н•ң л¶Җл¶„м—җ div нғңк·ёлҘј л§Ңл“Өкі  н•ҙлӢ№ нғңк·ём—җ көҗм°ЁмӢңм—җ лӢӨмқҢ нҺҳмқҙм§Җ fetchлҘј н•ҳлҸ„лЎқ м„Өм •н•ҳмҳҖлӢӨ.

м „лһөмқ„ м„ёмӣ мңјлӢҲ л°”лЎң hookмқ„ мғқм„ұн–ҲлӢӨ.

import { useEffect, useRef } from 'react';

const useInfiniteScroll = (data, FetchNextPage) => {
  const targetRef = useRef();

  const observer = new IntersectionObserver(
    ([entry], observer) => {
      if (entry.isIntersecting) {
        observer.unobserve(entry.target);
        FetchNextPage();
      }
    },
    { threshold: 0.8 }
  );

  useEffect(() => {
    if (!data) FetchNextPage();

    if (targetRef.current) observer.observe(targetRef.current);
    return () => observer && observer.disconnect();
  }, [data]);

  return { targetRef };
};

export default useInfiniteScroll;

к°ҖмғҒмқҳ divнғңк·ём—җ мӨ„ observerлҘј мғқм„ұнӣ„ refк°’мқ„ return н•ҙ к·ё к°’мқ„ к°ҖмғҒмқҳ divнғңк·ём—җ л„Јм–ҙ көҗм°Ё мӢңм—җ FetchNextPageк°Җ мӢӨн–үлҗҳлҸ„лЎқ hookмқ„ кө¬м„ұн•ҳмҳҖлӢӨ.

л¬ём ңм җ л°ңмғқ

н•ҳм§Җл§Ң мқҙ л°©лІ•мқҖ мҠӨнҒ¬лЎӨмқ„ кі„мҶҚ лӮҙлҰҙмҲҳлЎқ лҚ°мқҙн„°лҘј кі„мҶҚ л¶Ҳлҹ¬мҷ”кі  лҚ°мқҙн„°к°Җ мҲҳмӢӯк°ң мҲҳл°ұ к°ң лҗҳмһҗ н”„л Ҳмһ„мқҙ л–Ём–ҙм§ҖлҠ” л¬ём ңк°Җ л°ңмғқн–ҲлӢӨ. мқён”јлӢҲнӢ° мҠӨнҒ¬лЎӨмқ„ н•  л•Ңл§ҲлӢӨ DOMнҠёлҰ¬м—җ лҚ°мқҙн„° кІ°кіј к°’мқ„ м¶”к°Җн•ҙм•ј н•ҳлӢҲ div нғңк·ёк°Җ л§Һм•„м§Ҳ мҲҳл°–м—җ м—Ҷм—ҲлҚҳ кІғмқҙлӢӨ.

мғҲлЎңмҡҙ м „лһөВ 

мқҙлҘј н•ҙкІ°н•ҳкё° мң„н•ҙм„ң мғқк°Ғн•ҙ лӮёкІғмқҙ вҖҳмӮ¬мҡ©мһҗм—җкІҢ ліҙмқҙлҠ” л¶Җл¶„л§Ң elementлҘј л§Ңл“Өкі  ліҙмқҙм§Җ м•ҠлҠ” л¶Җл¶„мқҖ DOMм—җм„ң elementлҘј м ңкұ°н•ҳлҠ” нҳ•мӢқмңјлЎң л§Ңл“ӨмһҗвҖҷмҳҖлӢӨ н•ҳм§Җл§Ң к·ёлғҘ DOMм—җм„ң м ңкұ°лҘј н•ҳкІҢ лҗҳл©ҙ м ңкұ°лҗң DOMмқҳ нҒ¬кё°л§ҢнҒј нҷ”л©ҙмқҳ лҶ’мқҙк°Җ мӨ„м–ҙл“ӨкІҢ лҗҳм–ҙм„ң л¶Җмһҗм—°мҠӨлҹ¬мҡҙ кІ°кіјлҘј лӮікІҢ лҗңлӢӨ. л”°лқјм„ң м¶”к°Җм Ғмқё лЎңм§Ғмқҙ н•„мҲҳлЎң лҸҷл°ҳлҗҳм–ҙм•ј н–Ҳкі  мқҙ л°©лІ•мқ„ мҳӨлҠҳмқҳ м§‘ нҺҳмқҙм§Җм—җм„ң мӮ¬мҡ©н•ҳкі  мһҲм—ҲлӢӨ.

мӮ¬мҡ©мһҗк°Җ ліҙмқҙлҠ” viewмқҳ мқјм • л¶Җл¶„л§Ң domм—җ лӮЁкІЁл‘җкі  padding-topмқ„ мқҙмҡ©н•ҙ мӮ¬лқјм§„ л¶Җл¶„л§ҢнҒј кіөл°ұмқ„ л§Ңл“Өм–ҙм„ң domмқ„ кҙҖлҰ¬н•ҳкі  мһҲлҠ”кұё ліј мҲҳ мһҲлӢӨ.

мқҙлҹ¬л©ҙ мӮ¬мҡ©мһҗ мһ…мһҘм—җм„ңлҠ” лҚ°мқҙн„°лҘј лҒҠнӮҙм—Ҷмқҙ л¶Ҳлҹ¬мҳӨл©ҙм„ң ліј мҲҳ мһҲлҠ” мһҘм җмқҙ мһҲкі  лҚ°мқҙн„°лҘј м•„л¬ҙлҰ¬ л¶Ҳлҹ¬мҷҖлҸ„ domмһҗмІҙм—җлҠ” н• лӢ№лҗң м–‘л§ҢнҒјмқҳ div нғңк·ёк°Җ мһҲкё° л•Ңл¬ём—җ н”„л Ҳмһ„ л“ңлЎӯмқҙ мқјм–ҙлӮҳм§Җ м•ҠлҠ”лӢӨ. лӮҳлҸ„ мқҙ л°©лІ•мқ„ нҶ лҢҖлЎң лҳ‘к°ҷмқҙ л¬ём ңлҘј н•ҙкІ° н•ҙ ліҙкі мһҗ н•ҳмҳҖлӢӨ.

import React from 'react';
import PropTypes from 'prop-types';
import AutoSizer from 'react-virtualized-auto-sizer';
// мһҗлҸҷмңјлЎң widthлӮҳ heightмқ„ к°Җм ёмҳҙ
import { Grid, InfiniteLoader, WindowScroller } from 'react-virtualized';
// InfiniteLoader: л¬ҙн•ңмҠӨнҒ¬лЎӨ, WindowScroller: лёҢлқјмҡ°м Җмқҳ window мҠӨнҒ¬лЎӨл°”лҘј мӮ¬мҡ©н•ҳл Өл©ҙ мӮ¬мҡ©,

const InfiniteGridWrapper = ({
  data,
  hasNextPage,
  isNextPageLoading,
  loadNextPage,
  totalHeight,
  totalWidth,
  rowHeight,
  columnWidth,
  columnCount,
  itemComponent,
  isUseWindowScroll = false
}) => {
  const rowCount = hasNextPage ? data.length / 2 + 1 : data.length / 2 - 1;
  // лӢӨмқҢнҺҳмқҙм§Җк°Җ мһҲлӢӨл©ҙ лЎңл”©м°Ҫ(мҠӨмјҲл ҲнҶӨ UI)мқ„ мғқм„ұн•ҙм•јн•Ё (+2к°ң мғқм„ұ)
  // react-queryлҠ” л§Ҳм§Җл§ү лҚ°мқҙн„°лҘј к°Җм ёмҷ”лҠ”лҚ° к°’мқҙ м—Ҷмңјл©ҙ hasNextPageк°Җ falseк°Җ лҗҳлҜҖлЎң 0мқҙ лҗҳлҜҖлЎң н•ңлІҲ лҚ” к°Җм ёмҳЁ кјҙмқҙ лҗҳлӢҲ -2лЎң мЎ°м •н•ҙ мЈјм–ҙм•јн•Ё
  const loadMoreRows = isNextPageLoading ? () => {} : loadNextPage;
  // лҚ°мқҙн„°лҘј л¶Ҳлҹ¬мҳӨлҠ” лҸ„мӨ‘м—җлҠ” Fetchк°Җ мӢӨн–үлҗҳм§Җ м•ҠлҸ„лЎқ л§үмқҢ

  const isRowLoaded = (index) => !hasNextPage || index < data.length;
  // н•ҙлӢ№ indexк°Җ л°ӣм•„мҳЁ лҚ°мқҙн„°мқҳ лІ”мң„м•Ҳм—җ л“Өм–ҙк°ҖлҠ” нҷ•мқё
  // л°ӣм•„мҳЁ лҚ°мқҙн„°ліҙлӢӨ нҒ¬л©ҙ falseлҘј return

  return (
    <InfiniteLoader
      rowCount={rowCount}
      loadMoreRows={loadMoreRows}
      isRowLoaded={isRowLoaded}
      threshold={1}
    >
      {({ onRowsRendered }) => {
        const onSectionRendered = ({ rowStartIndex, rowStopIndex, columnStopIndex }) => {
          const startIndex = rowStartIndex * columnStopIndex;
          // 2м°Ёмӣҗмқ„ gridлҘј 1м°ЁмӣҗмңјлЎң ліҖнҷҳ
          const stopIndex = rowStopIndex * columnStopIndex;

          if (stopIndex >= data.length / 2 - 2)
            // лёҢлқјмҡ°м Җк°Җ ліҙм—¬м§ҖлҠ” кіімқҙ л°”лӢҘліҙлӢӨ 1м№ё мң„лқјл©ҙ лӢӨмқҢлҚ°мқҙн„°лҘј л¶Ҳлҹ¬мҳҙ
            onRowsRendered({
              startIndex,
              stopIndex
            });
        };
        return isUseWindowScroll ? (
          // лёҢлқјмҡ°м Җмқҳ window мҠӨнҒ¬лЎӨмқ„ мӮ¬мҡ©н•ҳл Өл©ҙ WindowScroller wrapperк°Җ мһҲм–ҙм•јн•ҳкі  мӮ¬мҡ©н•ҳм§Җ м•Ҡмңјл Өл©ҙ wrapperк°Җ н•„мҡ”м—ҶмқҢ
          <AutoSizer className="autoSizer">
            {({ height: rowDefaultHeight, width }) => (
              <WindowScroller class="windowScroller">
                {({ height, scrollTop }) => {
                  return (
                    <Grid
                      autoHeight={isUseWindowScroll}
                      width={totalWidth ?? width}
                      height={totalHeight ?? height}
                      columnWidth={columnWidth}
                      rowHeight={rowHeight ?? rowDefaultHeight}
                      rowCount={rowCount}
                      columnCount={columnCount}
                      scrollTop={scrollTop}
                      onSectionRendered={onSectionRendered}
                      cellRenderer={({ rowIndex, columnIndex, style, key }) =>
                        itemComponent({
                          isRowLoaded,
                          rowIndex,
                          columnIndex,
                          style,
                          key
                        })
                      }
                    />
                  );
                }}
              </WindowScroller>
            )}
          </AutoSizer>
        ) : (
          <Grid
            width={totalWidth}
            height={totalHeight}
            columnWidth={columnWidth}
            rowHeight={rowHeight}
            rowCount={rowCount}
            columnCount={columnCount}
            onSectionRendered={onSectionRendered}
            cellRenderer={({ rowIndex, columnIndex, style, key }) =>
              itemComponent({
                isRowLoaded,
                rowIndex,
                columnIndex,
                style,
                key
              })
            }
          />
        );
      }}
    </InfiniteLoader>
  );
};

InfiniteGridWrapper.propTypes = {
  hasNextPage: PropTypes.bool,
  data: PropTypes.array,
  isNextPageLoading: PropTypes.bool,
  loadNextPage: PropTypes.func,
  totalHeight: PropTypes.number,
  totalWidth: PropTypes.number,
  rowHeight: PropTypes.number,
  columnWidth: PropTypes.number,
  columnCount: PropTypes.number,
  itemComponent: PropTypes.func,
  isUseWindowScroll: PropTypes.bool
};

export default InfiniteGridWrapper;
